<script>
const scene = document.getElementById("scene");

/* RAMAS SUAVES */
setTimeout(()=>{
    createBranch(-20, -40);
    createBranch(20, 40);
},3500);

function createBranch(offset, angle){
    let b = document.createElement("div");
    b.className="branch";
    b.style.bottom="260px";
    b.style.left="50%";
    b.style.transform=`translateX(${offset}px) rotate(${angle}deg)`;
    scene.appendChild(b);

    setTimeout(()=>{
        b.style.animation="growBranch 2s ease-out forwards";
    },100);
}

/* FUNCIÓN CORAZÓN */
function heart(t){
    return {
        x:16*Math.pow(Math.sin(t),3),
        y:13*Math.cos(t)-5*Math.cos(2*t)-2*Math.cos(3*t)-Math.cos(4*t)
    };
}

/* GIRASOLES REALISTAS */
setTimeout(()=>{
    let total = 350;

    for(let i=0;i<total;i++){
        let t = (i/total)*Math.PI*2;
        let p = heart(t);

        let f = document.createElement("div");
        f.className="sunflower";

        f.style.left = (window.innerWidth/2 + p.x*10) + "px";
        f.style.top = (window.innerHeight/2 - p.y*10) + "px";

        scene.appendChild(f);

        setTimeout(()=>{
            f.style.opacity=1;
            f.style.transform="scale(1)";
        }, i*10);
    }

    showCard();

},5500);

/* MOSTRAR MENSAJE */
function showCard(){
    setTimeout(()=>{
        document.getElementById("card").classList.add("show");
        startTimer();
    },3000);
}

/* CONTADOR */
function startTimer(){
    const start = new Date("2023-06-05");

    setInterval(()=>{
        let now = new Date();
        let diff = now - start;

        let d = Math.floor(diff/86400000);
        let h = Math.floor(diff/3600000)%24;
        let m = Math.floor(diff/60000)%60;

        document.getElementById("timer").innerText =
        `${d} días ${h}h ${m}m juntos 💛`;
    },1000);
}
